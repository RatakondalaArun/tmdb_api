Future<Map> removeItem(String? sessionId, String? listId, int? mediaId) {
  return _v._query(
    '$_endPoint/${listId.toString()}/remove_item',
    method: HttpMethod.POST,
    postBody: {'media_id': '$mediaId'},
    optionalQueries: ['session_id=$sessionId'],
  );
}
