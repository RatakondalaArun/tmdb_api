Future<Map> getTvList({
  String? endDate,
  String? startDate,
  int page = 1,
}) {
  final para = <String>[];
  if (endDate != null) {
    para.add('end_date=$endDate');
  }
  if (startDate != null) {
    para.add('start_date=$startDate');
  }
  para.add('page=${page < 1 || page > 1000 ? 1 : page}');

  return _v._query('tv/$_endPoint', optionalQueries: para);
}
