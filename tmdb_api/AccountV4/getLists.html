Future<Map> getLists(
  String accessToken,
  String accountId, {
  int page = 1,
}) {
  if (page < 1 || page > 1000) {
    throw ArgumentError('page < 1 || page > 1000 is true');
  }

  return _v._query(
    '$_endPoint/$accountId/lists',
    postHeaders: {
      'Authorization': 'Bearer $accessToken',
      'Content-Type': 'application/json;charset=utf-8',
    },
  );
}
