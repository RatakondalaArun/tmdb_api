Future<Map> getRatedMovies(
  String sessionId,
  int accountId, {
  String? language,
  int page = 1,
  SortBy sortBy = SortBy.createdAtAsc,
}) {
  if (accountId < 1 || page < 1 || page > 1000) {
    throw ArgumentError('accountId < 1 || page < 1 || page > 1000 is true');
  }
  String sort;
  if (sortBy == SortBy.createdAtAsc) {
    sort = 'created_at.asc';
  } else {
    sort = 'created_at.desc';
  }

  return _v._query(
    '$_endPoint/$accountId/rated/movies',
    optionalQueries: [
      'session_id=$sessionId',
      'language=${language ?? _v._tmdb.defaultLanguage}',
      'page=$page',
      'sort_by=$sort',
    ],
  );
}
